int isprime(int);
main()
{
    int i,c=0;
    for(i=2;i<=100;i++)
    {
        if(isprime(i))
        {
            printf("%4d",i);
            c++;
            if(c%10==0)printf("\n");
        }
    }
    printf("\n");
}
int isprime(int n)
{
    int i;
    for(i=2;i<=sqrt(n);i++)
        if(n%i==0)return 0;
    return 1;
}
